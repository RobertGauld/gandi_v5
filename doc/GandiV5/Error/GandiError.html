def self.from_hash(hash)
  hash['errors'] ||= []

  new(
    (hash['errors'].count > 1 ? "\n" : '') +
      hash['errors'].map { |err| "#{err['location']}->#{err['name']}: #{err['description']}" }
                         .join("\n")
  )
end
